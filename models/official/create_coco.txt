export STORAGE_BUCKET=gs://deep-mapan-train
export DATA_DIR=${STORAGE_BUCKET}/coco2017

export TRAIN_IMAGE_DIR=${DATA_DIR}/image
export TRAIN_IMAGE_INFO_FILE=${DATA_DIR}/imgae_info

export TRAIN_ANNOTATIONS_FILE=${DATA_DIR}/train_annotations
export CAPTION_ANNOTATIONS_FILE=${DATA_DIR}/caption_annotations

export OUTPUT_DIR=${DATA_DIR}/output
export FILE_PREFIX=2017

python create_coco_tf_record.py --logtostderr \
      --image_dir="${TRAIN_IMAGE_DIR}" \
      --image_info_file="${TRAIN_IMAGE_INFO_FILE}" \
      --object_annotations_file="${TRAIN_ANNOTATIONS_FILE}" \
      --caption_annotations_file="${CAPTION_ANNOTATIONS_FILE}" \
      --output_file_prefix="${OUTPUT_DIR/FILE_PREFIX}" \
      --num_shards=100
